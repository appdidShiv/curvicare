Login

This endpoint is used to authenticate user by token or register new user if user is not found in database it will create new user and return it with token if user is found in database it will return old user with new token

Example request:
curl --request POST \
    "http://localhost/api/v1/user/login" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"token\": \"<<json web token>>\",
    \"device_id\": \"1e6c1d3c2f5b4f6d8c3e5a7b9c6d8f5e\"
}"

const url = new URL(
    "http://localhost/api/v1/user/login"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "token": "<<json web token>>",
    "device_id": "1e6c1d3c2f5b4f6d8c3e5a7b9c6d8f5e"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
